#Для скачивания образов docker необходимо подключить машину к интернету, при этом необходимо отключить интерфейс локальной сети:
ifconfig <название интерфейса> down
#Чтобы включить интерфейс меняем down на up - ifconfig ens33 up

#Добавляем docker в автозапуск:
systemctl enable docker.service

#Создание тома "dbvolume" для хранения инфы:
docker volume create dbvolume

#Запуск docker-compose файла "wiki.yaml" без блокировки консоли:
docker-compose -f wiki.yaml up -d
#Для принудительной остановки контейнеров используем команду:
docker compose -f wiki.yaml down

#После установки "mediawiki" скачиваем файл "LocalSettings.php" и копируем его в директорию пользователя (там где должен лежать "wiki.yaml" файл). 
#Далее редактируем файл "wiki.yaml" - удаляем символ комментария перед строчкой "- ./LocalSettings.php:/var/www/html/LocalSettings.php".
#Перезапускаем контейнеры.

#Контейнеры уже настроены на постоянный запуск после старта системы. Так что остается только запустить один раз контейнеры и спать будет хорошо.

#Порт может быть занят IPA. У меня так было. Вырубите её, поднимите контейнеры, если не встают. И потом включите Ипу.
