#Если часовой пояс на сервере не Московский (проверяем с помощью команды timedatectl), то вводим эту гадость.
timedatectl set-timezone Europe/Moscow

#После раскопирования конфигов не забываем перезагружать сервис.
systemctl restart chronyd

#Для проверки того, подключилося ли всё к серверу времени используем эту команду.
chronyc sources

#Для проверки того, кто подключён к серверу. Клиенты выводятся постепенно, так что не пугайтесь.
chronyc clients
